setup a plotting function that is flexible for grouping and faceting designed with age/length by year or depth, location etc. but we’ll use it with car data…

#' plot biological features by groups w/facets
#'
#' @param data dataframe (recommend everything be lowercase)
#' @param x x value for fig (e.g., "age")
#' @param y y value for fig (e.g., "length")
#' @param group grouping variable (e.g., "sex")
#' @param group_fct turn the grouping variable into a factor? default FALSE
#' @param facet facetting variable (e.g., "strata")
#'
#' @return
#' @export
#'
#' @examples
plot_biology <- function(data, x, y, group = NULL, group_fct = FALSE, facet = NULL) {
  
   # set view
  ggplot2::theme_set(afscassess::theme_report())
  
  
  if(!is.null(facet)) {
    if(!is.null(group) & isTRUE(group_fct)) {
      # discrete group color
      data %>% 
        tidytable::mutate.(group = factor(.data[[group]])) %>% 
        ggplot2::ggplot(ggplot2::aes(.data[[x]], .data[[y]], color = group)) +
        ggplot2::geom_point() + 
        ggplot2::facet_wrap(~.data[[facet]])
    } else if(!is.null(group) & isFALSE(group_fct)) {
      # continuous group color
      data %>% 
        ggplot2::ggplot(ggplot2::aes(.data[[x]], .data[[y]], color = .data[[group]])) +
        ggplot2::geom_point() + 
        ggplot2::facet_wrap(~.data[[facet]])
    } else {
      data %>% 
        ggplot2::ggplot(ggplot2::aes(.data[[x]], .data[[y]])) +
        ggplot2::geom_point() + 
        ggplot2::facet_wrap(~.data[[facet]])
    }
    
  } else {
    if(!is.null(group) & isTRUE(group_fct)) {
      # discrete group color
      data %>% 
        tidytable::mutate.(group = factor(.data[[group]])) %>% 
        ggplot2::ggplot(ggplot2::aes(.data[[x]], .data[[y]], color = group)) +
        ggplot2::geom_point()
    } else if(!is.null(group) & isFALSE(group_fct)) {
      # continuous group color
      data %>% 
        ggplot2::ggplot(ggplot2::aes(.data[[x]], .data[[y]], color = .data[[group]])) +
        ggplot2::geom_point()
    } else {
      data %>% 
        ggplot2::ggplot(ggplot2::aes(.data[[x]], .data[[y]])) +
        ggplot2::geom_point()
    }
  }
  
}

Then it is just a matter of turning switches on and off:

plot_biology(dat, "hp", "mpg")

plot_biology(dat, "hp", "mpg", group="hp")

plot_biology(dat, "hp", "mpg", group="hp", group_fct=TRUE)

plot_biology(dat, "hp", "mpg", group="hp", facet="cyl")

Nice part is that you can change things up “post-function” since it is outputting a ggplot object.

plot_biology(dat, "hp", "mpg", group="hp", facet="cyl") +
  scico::scale_color_scico(palette = 'roma') +
  ggplot2::xlab("Horse power") +
  ggplot2::ylab("Miles / Gallon") +
  ggplot2::theme(legend.justification=c(1,0),
                 legend.position=c(0.93,0.6))

plot_biology(dat, "hp", "mpg", group="gear", group_fct = TRUE) +
  scico::scale_color_scico_d(palette = 'roma') +
  ggplot2::stat_smooth(method=  "gam", formula = y~s(x, k=3)) +
  ggplot2::xlab("Horse power") +
  ggplot2::ylab("Miles / Gallon") +
  ggplot2::theme(legend.justification=c(1,0),
                 legend.position=c(0.93,0.6))
